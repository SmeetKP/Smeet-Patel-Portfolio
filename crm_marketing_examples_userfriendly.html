Sample Data

import pandas as pd
users = pd.DataFrame({
    'user_id': [1, 2, 3, 4],
    'email_opt_in': [True, False, True, True],
    'last_active_days': [2, 45, 10, 5]

Campaign Targeting Visualization

import matplotlib.pyplot as plt

Churn Prediction Example

from sklearn.linear_model import LogisticRegression
X = users[['last_active_days']]
y = users['segment'].apply(lambda x: 1 if x == 'Lapsed' else 0)
model = LogisticRegression().fit(X, y)
users['churn_prob'] = model.predict_proba(X)[:,1]
users[['user_id', 'churn_prob']]
